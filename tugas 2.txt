# Validasi nama
nama_benar = "rifa"  # ganti sesuai kebutuhan

while True:
    nama = input("Masukkan nama anda: ")
    
    if nama == nama_benar:
        print("Nama benar, lanjut ke program berikutnya.\n")
        break
    else:
        print("Silahkan coba lagi.\n")

# Input angka untuk tabel perkalian
angka = int(input("Masukkan angka (1-10): "))

# Validasi angka
if angka >= 1 and angka <= 10:
    print(f"\nTabel perkalian {angka}:")
    for i in range(1, 11):
        print(f"{angka} x {i} = {angka * i}")
else:
    print("Angka harus antara 1 sampai 10.")
